Membuat direktori baru > mkdir
Menampilkan direktori/file > ls
Masuk kedalam direktori > cd
Kembali ke direktori sebelumnya > cd ..
Mengganti nama direktori > mv
Membuat file > cat
Menghapus direktori > rmdir
Melakukan update > sudo apt-get update
Menghapus command yang ada di terminal > clear
Mengecek versi kernel > uname -a

root@rindha:~# mkdir coba

root@rindha:~# ls
 coba
  
root@rindha:~# cat coba

root@rindha:~# cd coba

root@rindha:~/coba# ls

root@rindha:~/coba# cd ..

root@rindha:~# mv coba newcoba
newcoba   

root@rindha:~# rmdir newcoba

root@rindha:~# sudo apt-get update
